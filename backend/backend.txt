# Coding Standards

Always:

* Use TypeScript
* Use async/await
* Use try/catch
* Validate all inputs
* Never trust uploaded files
* Keep controllers small
* Keep business logic in services
* Use environment variables
* Write meaningful logs


Every service must log:

* Requests
* Errors
* Warnings
* Job completion

Never log:

# logger.ts

```ts
import winston from 'winston';

const logger = winston.createLogger({
  level: 'info',

  format: winston.format.combine(
    winston.format.timestamp(),
    winston.format.errors({ stack: true }),
    winston.format.json()
  ),

  defaultMeta: {
    service: 'pdfkit-backend'
  },

  transports: [
    new winston.transports.Console(),

    new winston.transports.File({
      filename: 'logs/error.log',
      level: 'error'
    }),

    new winston.transports.File({
      filename: 'logs/combined.log'
    })
  ]
});

export default logger;
```

# Merge PDF Service Example

```ts
import { PDFDocument } from 'pdf-lib';
import fs from 'fs';

export const mergePDFs = async (files: string[]) => {
  const mergedPdf = await PDFDocument.create();

  for (const file of files) {
    const pdfBytes = fs.readFileSync(file);

    const pdf = await PDFDocument.load(pdfBytes);

    const pages = await mergedPdf.copyPages(
      pdf,
      pdf.getPageIndices()
    );

    pages.forEach((page) => mergedPdf.addPage(page));
  }

  return mergedPdf.save();
};
```

# File Upload Middleware

```ts
  destination: 'uploads/',
```

# DOCX → PDF Conversion

```ts
import { exec } from 'child_process';

export const convertToPDF = (
  inputPath: string,
  outputDir: string
) => {
  return new Promise((resolve, reject) => {
    const command = `libreoffice --headless --convert-to pdf ${inputPath} --outdir ${outputDir}`;

    exec(command, (error) => {
      if (error) {
        return reject(error);
      }

      resolve(true);
    });
  });
};
```

# 13. Queue Service

# Why Queues Matter

Without queues:

* server blocks
* requests timeout
* app crashes under load

---

# Install

```bash
npm install bullmq ioredis
```

---

# Queue Setup

```ts
import { Queue } from 'bullmq';
import Redis from 'ioredis';

const connection = new Redis();

export const pdfQueue = new Queue('pdf-jobs', {
  connection
});
```

---

# Add Job

```ts
await pdfQueue.add('merge-pdf', {
  files
});
```

---

# Worker Example

```ts
import { Worker } from 'bullmq';

const worker = new Worker(
  'pdf-jobs',

  async (job) => {
    logger.info(`Processing ${job.name}`);
  },

  { connection }
);
```

# 15. Global Error Handler

# middleware/errorHandler.ts

```ts
import { Request, Response, NextFunction } from 'express';

export const errorHandler = (
  error: any,
  req: Request,
  res: Response,
  next: NextFunction
) => {
  logger.error(error.message);

  return res.status(500).json({
    success: false,
    message: 'Internal Server Error'
  });
};
```
