QUICK REFERENCE — Commands You'll Use Every Day
=================================================

SSH INTO SERVER
  ssh -i pdfkit-server.key ubuntu@YOUR_PUBLIC_IP

NAVIGATE TO PROJECT
  cd ~/PDFkit/backend


STATUS & HEALTH
  docker-compose ps                          # all container status
  ~/check-health.sh                          # all service health endpoints
  docker stats --no-stream                   # CPU + RAM per container
  df -h /                                    # disk usage


LOGS
  docker-compose logs -f                     # all services live
  docker-compose logs -f pdf-service         # single service live
  docker-compose logs --tail=50 api-gateway  # last 50 lines
  tail -f logs/api-gateway-error.log         # error log file


START / STOP
  docker-compose up -d                       # start all
  docker-compose down                        # stop all (data kept)
  docker-compose restart api-gateway         # restart one service
  docker-compose restart                     # restart all


UPDATE CODE
  git pull origin main
  docker-compose up --build -d


REBUILD ONE SERVICE
  docker-compose up --build -d pdf-service


ENTER A CONTAINER
  docker-compose exec pdf-service sh
  docker-compose exec mysql bash
  docker-compose exec storage-service sh


MYSQL
  docker-compose exec mysql mysql -u root -pYOUR_PASSWORD pdfkit
  docker-compose exec mysql mysqldump -u root -pYOUR_PASSWORD pdfkit > backup.sql


NGINX
  sudo nginx -t                              # test config
  sudo systemctl reload nginx               # reload config
  sudo systemctl status nginx               # check status
  sudo tail -f /var/log/nginx/error.log     # nginx errors


SSL
  sudo certbot certificates                 # check expiry
  sudo certbot renew                        # renew manually


DISK CLEANUP
  docker system prune -f                    # remove unused images/containers
  find logs/ -name "*.log" -mtime +7 -delete  # delete old logs
